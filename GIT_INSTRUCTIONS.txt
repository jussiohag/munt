# GIT INSTRUCTIONS - NEW REPO

# Import project
git clone https://github.com/munt/munt


# Create a new repository on the github
--> github --> new repo --> munt

# Create a new repository on the command line
cd ~\Desktop\git\munt
git init

# Add project description (optional?)
echo "# New project" >> README.md
git add README.md
git commit -m "first commit"
git branch -M main

# Push to github
git remote add origin git@github.com:jussiohag/munt.git
git push -u origin main
